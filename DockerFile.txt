# --- Runtime image
FROM mcr.microsoft.com/dotnet/aspnet:9.0 AS base
WORKDIR /app
EXPOSE 8080
ENV ASPNETCORE_URLS=http://+:8080

# --- Build image
FROM mcr.microsoft.com/dotnet/sdk:9.0 AS build
WORKDIR /src

# copy only project files first for layer caching
COPY ["cs-project/cs-project.csproj", "cs-project/"]
COPY ["cs-project.Core/cs-project.Core.csproj", "cs-project.Core/"]
COPY ["cs-project.Infrastructure/cs-project.Infrastructure.csproj", "cs-project.Infrastructure/"]
COPY ["cs-project.Tests/cs-project.Tests.csproj", "cs-project.Tests/"]
RUN dotnet restore "cs-project/cs-project.csproj"

# copy the rest of the source
COPY . .
WORKDIR /src/cs-project
RUN dotnet build "cs-project.csproj" -c Release -o /app/build

# --- Publish
FROM build AS publish
RUN dotnet publish "cs-project.csproj" -c Release -o /app/publish /p:UseAppHost=false

# --- Final image
FROM base AS final
WORKDIR /app
COPY --from=publish /app/publish .
ENTRYPOINT ["dotnet", "cs_project.dll"]
